-- 1. Tabla de progreso de ejercicios (definicion)
CREATE TABLE IF NOT EXISTS definicion_workout_progress (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  exercise_id TEXT NOT NULL,
  day TEXT NOT NULL,
  week INTEGER NOT NULL,
  weights JSONB DEFAULT '[]'::jsonb,
  series_count INTEGER DEFAULT 3,
  date TIMESTAMPTZ DEFAULT NOW(),
  is_alternative BOOLEAN DEFAULT FALSE,
  alternative_index INTEGER,
  observations TEXT DEFAULT '',
  rpe_actual DECIMAL,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 2. Tabla de listas de compras (definicion)
CREATE TABLE IF NOT EXISTS definicion_shopping_lists (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  selected_weeks JSONB DEFAULT '[]'::jsonb,
  selected_days JSONB DEFAULT '[]'::jsonb,
  items JSONB DEFAULT '[]'::jsonb,
  generated_date TIMESTAMPTZ DEFAULT NOW(),
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 3. Tabla de ajustes/settings (definicion)
CREATE TABLE IF NOT EXISTS definicion_settings (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  setting_key TEXT UNIQUE NOT NULL,
  setting_value JSONB,
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 4. Tabla de composicion corporal (NUEVA)
CREATE TABLE IF NOT EXISTS definicion_body_composition (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  week INTEGER NOT NULL,
  date TIMESTAMPTZ DEFAULT NOW(),
  peso DECIMAL NOT NULL,
  grasa_corporal DECIMAL,
  cintura DECIMAL,
  cadera DECIMAL,
  pecho DECIMAL,
  brazo DECIMAL,
  muslo DECIMAL,
  notas TEXT DEFAULT '',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 5. Tabla de registros de cardio (NUEVA)
CREATE TABLE IF NOT EXISTS definicion_cardio_logs (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  day TEXT NOT NULL,
  week INTEGER NOT NULL,
  tipo TEXT NOT NULL,
  duracion_minutos INTEGER DEFAULT 0,
  completado BOOLEAN DEFAULT FALSE,
  notas TEXT DEFAULT '',
  date TIMESTAMPTZ DEFAULT NOW(),
  created_at TIMESTAMPTZ DEFAULT NOW()
);
